settings

	background 					
		: Sheet - Background color
	foreground 					
		: Sheet - Foreground color	
	caret 						
		: Caret - Foreground color
	invisibles 					
		: Invisible chars - Foreground color
		- ( tabs, whitespaces, etc. )
		- ( Ignored )

	lineHighlight 				
		: Line Highlight - Background color
	selection
		: Highlight color (dimmed)
	selectionBackground
		: Background color
	selectionBorder
		: Border color
	inactiveSelection
		: Highlight color

	findHighlight 				Find - Background color of Matches
	findHighlightForeground 	Find - Foreground color of Matches

	highlight 					Regions added via sublime.add_regions with sublime.DRAW_OUTLINED
	highlightForeground 		

	gutter 						Gutter - Background color
	gutterForeground 			Gutter - Foreground color

	guide 						Guide - Foreground color (tab guide, ...)
	stackGuide 					
	activeGuide 				

	bracketContentsOptions 		Values: underline, ...
	tagsForeground 				Tags - Foreground color
	tagsOption 					Tags - Decoration 			Values: underline

	shadow 						Sheet border shadow (minimap to the right and gutter to the left)
	shadowWidth
